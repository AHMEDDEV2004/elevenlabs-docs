```Bash Curl
curl -X 'DELETE' \
  'https://api.elevenlabs.io/v1/history/<history-item-id>' \
  --header 'accept: application/json' \
  --header 'xi-api-key: <xi-api-key>'
```

```Python Python
import requests

url = "https://api.elevenlabs.io/v1/history/<history-item-id>"

headers = {
  "Accept": "audio/json",
  "xi-api-key": "<xi-api-key>"
}

response = requests.delete(url, headers=headers)

print(response.text)
```

```Javascript Javascript
async function Delete_History_Item() {
    return fetch('https://api.elevenlabs.io/v1/history/<history-item-id>', {
        method: 'DELETE',
        headers: {
            'accept': 'application/json',
            'xi-api-key': '<xi-api-key>'
        }
    }).then(response => response.text()).then(text => JSON.parse(text))
}

response = await Delete_History_Item()
console.log(response)
```
