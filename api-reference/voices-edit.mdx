```Bash Curl
curl -X 'POST' \
  'https://api.elevenlabs.io/v1/voices/<voice-id>/edit' \
  -H 'accept: application/json' \
  -H 'xi-api-key: <xi-api-key>' \
  -H 'Content-Type: multipart/form-data' \
  -F 'name=New Name' \
  -F 'files=@sample.mp3;type=audio/mpeg' \
  -F 'labels={"accent":"American"}'
```

```Python Python
import requests

url = "https://api.elevenlabs.io/v1/voices/<voice-id>/edit"

headers = {
  "Accept": "application/json",
  "xi-api-key": "<xi-api-key>"
}

data = {
    'name': 'Voice New name',
    'labels': '{"accent": "British"}'
}

files = [
    ('files', ('sample1.mp3', open('sample1.mp3', 'rb'), 'audio/mpeg')),
    ('files', ('sample2.mp3', open('sample2.mp3', 'rb'), 'audio/mpeg'))
]

response = requests.post(url, headers=headers, data=data, files=files)
print(response.text)
```
