```Python Python
import requests

url = "https://api.elevenlabs.io/v1/voices/settings/default"

headers = {
  "Accept": "application/json"
}

response = requests.get(url, headers=headers)

print(response.text)
```

```Javascript Javascript
async function Get_DefaultVoiceSettings() {
    return fetch('https://api.elevenlabs.io/v1/voices/settings/default', {
        headers: {
            'accept': 'application/json',
            'xi-api-key': '<xi-api-key>'
        }
    }).then(response => response.text()).then(text => JSON.parse(text))
}

response = await Get_DefaultVoiceSettings()
console.log(response)
```
