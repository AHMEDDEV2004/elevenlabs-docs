```Bash Curl
curl -X 'POST' \
  'https://api.elevenlabs.io/v1/voices/<voice-id>/settings/edit' \
  --header 'accept: application/json' \
  --header 'xi-api-key: <xi-api-key>' \
  --header 'Content-Type: application/json' \
  --data '{
    "stability": 0.5,
    "similarity_boost": 0.5
  }'
```

```Python Python
import requests

url = "https://api.elevenlabs.io/v1/voices/<voice-id>/settings/edit"

headers = {
  "Accept": "application/json",
  "Content-Type": "application/json",
  "xi-api-key": "<xi-api-key>"
}

data = {
  "stability": 0.5,
  "similarity_boost": 0.5
}

response = requests.post(url, json=data, headers=headers)

print(response.text)
```

```Javascript javascript
async function Edit_Voice_Settings() {
    return fetch('"https://api.elevenlabs.io/v1/voices/<voice-id>/settings/edit', {
        headers: {
            'accept': 'application/json',
            'xi-api-key': '<xi-api-key>'
        },
        body: JSON.stringify({
            "stability": 0.5,
            "similarity_boost": 0.5
        })
    }).then(response => response.text()).then(text => JSON.parse(text))
}

response = await Edit_Voice_Settings()
console.log(response)
```
